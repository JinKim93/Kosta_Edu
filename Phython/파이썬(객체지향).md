### 자동차 클래스

```py
class MyCar:
    #자동차의 멤버변수
    color = ""
    maxspeed = 0
    direction = "" 

    #자동차의 기능
    def setSpeed(self,s_value):
        self.speed = s_value

    def getSpeed(self): #파이썬에서는 아무것도 전달할게 없으면 self이다
        return self.maxspeed

    def setDirection(self,s_value):
        self.direction = s_value

    def getDirection(self):
        return self.direction        

    def setColor(self,s_value):
        self.color = s_value

    def getColor(self):
        return self.color


mycar1 = MyCar() # 객체생성
mycar2 = MyCar()
mycar3 = MyCar() 
        
```
