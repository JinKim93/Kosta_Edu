# 정수를 입력 받아 점수범위에 맞게 등급을 출력하시오

```c
#include <stdio.h>

int main(void)
{
	int grade = 0;
	printf("성적을 입력하시오");
	scanf_s("%d", &grade);

	if (grade >= 91  && grade <=100)
	{
		printf("A학점 입니다\n");
	}
	else if (grade >=81 && grade <=90)
	{
		printf("B학점 입니다\n");
	}
	else if (grade >=71 && grade <=80)
	{
		printf("C학점 입니다\n");
	}
	else if (grade >=61 && grade <=70)
	{
		printf("C학점 입니다\n");
	}
	else if(grade > 0 && grade <=60)
	{
		printf("D학점 입니다\n");
	}
	else
	{
		printf("점수의 범위를 벗어났습니다.\n");
	}
}
```
